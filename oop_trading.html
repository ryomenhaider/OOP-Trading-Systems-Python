Encapsulation — the OrderBook problem

Encapsulation isn't just slapping underscores on attributes. It's about defining a clean interface and hiding the internal representation so you can swap it without breaking consumers.

__slots__ @property name mangling __repr__ / __str__
order.py — BAD (no encapsulation) 
# ❌ everything public, no invariants
class Order:
    def __init__(self, price, qty, side):
        self.price = price   # mutable!
        self.qty   = qty     # can be -1 externally
        self.side  = side    # can be "SELL" or "sell"
        self.filled = 0

# Caller does this and nothing stops it:
o = Order(100.5, 50, "buy")
o.qty = -999   # 💀 silent corruption
o.side = "BOTH"  # 💀 invalid state
order.py — GOOD 
from decimal import Decimal
from enum import Enum, auto

class Side(Enum):
    BID = auto()
    ASK = auto()

class Order:
    # ✅ No __dict__; fixed memory layout
    __slots__ = ('_price', '_qty',
                 '_side', '_filled')

    def __init__(self, price: Decimal,
                 qty: int, side: Side):
        self._price  = Decimal(price)
        self.qty     = qty    # through setter
        self._side   = side
        self._filled = 0

    @property
    def qty(self): return self._qty

    @qty.setter
    def qty(self, v):
        if v <= 0:
            raise ValueError(f"qty must be > 0, got {v}")
        self._qty = v

    @property
    def remaining(self):
        return self._qty - self._filled
Why __slots__ matters for an OrderBook: A live order book can hold thousands of Order objects. Without __slots__, each instance carries a full __dict__ (~200 bytes overhead). With slots, you drop to ~56 bytes per object. At 10,000 orders, that's 1.4 MB vs 200 KB. In a hot path this also means better cache locality.

orderbook.py 
from sortedcontainers import SortedDict
from decimal import Decimal
from typing import Optional

class OrderBook:
    """
    Encapsulates bid/ask ladders.
    Consumers NEVER touch _bids/_asks directly.
    """
    __slots__ = ('_bids', '_asks', '_symbol', '__imbalance_cache')
    #                                         ^ double underscore = name-mangled
    #                  → _OrderBook__imbalance_cache  (hard to access externally)

    def __init__(self, symbol: str):
        self._symbol = symbol
        self._bids   = SortedDict(lambda x: -x)  # desc: best bid first
        self._asks   = SortedDict()               # asc:  best ask first
        self.__imbalance_cache = None

    def update(self, side: Side, price: Decimal, qty: int) -> None:
        """Public interface: the ONLY way to mutate state."""
        book = self._bids if side == Side.BID else self._asks
        if qty == 0:
            book.pop(price, None)
        else:
            book[price] = qty
        self.__imbalance_cache = None  # invalidate cache on mutation

    @property
    def best_bid(self) -> Optional[Decimal]:
        return self._bids.keys()[0] if self._bids else None

    @property
    def best_ask(self) -> Optional[Decimal]:
        return self._asks.keys()[0] if self._asks else None

    @property
    def spread(self) -> Optional[Decimal]:
        b, a = self.best_bid, self.best_ask
        return (a - b) if b and a else None

    @property
    def imbalance(self, depth: int = 5) -> float:
        """
        OBI = (bid_vol - ask_vol) / (bid_vol + ask_vol)
        Cached until next update() call.
        """
        if self.__imbalance_cache is not None:
            return self.__imbalance_cache
        bv = sum(list(self._bids.values())[:depth])
        av = sum(list(self._asks.values())[:depth])
        total = bv + av
        self.__imbalance_cache = (bv - av) / total if total else 0.0
        return self.__imbalance_cache

    def __repr__(self):
        return (f"OrderBook({self._symbol!r} "
                f"bid={self.best_bid} ask={self.best_ask} "
                f"spread={self.spread})")
Key insight: The __imbalance_cache with double underscore triggers Python's name mangling → stored as _OrderBook__imbalance_cache. Subclasses won't accidentally stomp it. This is real production encapsulation, not just naming convention.

from abc import ABC, abstractmethod
from collections import deque
from dataclasses import dataclass
import time

class BaseDetector(ABC):
    def __init__(self, symbol: str, window: int = 100):
        self.symbol  = symbol
        self.window  = window
        self._scores: deque[float] = deque(maxlen=window)

    @abstractmethod
    def score(self, snapshot: dict) -> float:
        """Return a raw anomaly score [0, 1]."""

    def detect(self, snapshot: dict) -> Optional['Alert']:
        s = self.score(snapshot)
        self._scores.append(s)
        return None  # base: no alerting logic

    def _normalize(self, raw: float, lo=0.0, hi=1.0) -> float:
        return max(0.0, min(1.0, (raw - lo) / (hi - lo + 1e-9)))


# ─── Mixin (no __init__ state of its own → cooperative) ───────────
class AlertingMixin:
    alert_threshold: float = 0.75
    cooldown_ms:     int   = 5000
    _last_alert_ts:  float = 0.0

    def detect(self, snapshot: dict):
        # super() here follows MRO → calls SpoofingDetector.detect
        result = super().detect(snapshot)
        score  = self._scores[-1] if self._scores else 0.0
        return self._maybe_alert(score) or result

    def _maybe_alert(self, score: float):
        now = time.time() * 1000
        on_cooldown = (now - self._last_alert_ts) < self.cooldown_ms
        if score >= self.alert_threshold and not on_cooldown:
            self._last_alert_ts = now
            return Alert(symbol=self.symbol, score=score,
                         kind=self.__class__.__name__)
        return None


# ─── Concrete detector ────────────────────────────────────────────
class SpoofingDetector(BaseDetector):
    def __init__(self, symbol, cancel_threshold=0.8, **kw):
        super().__init__(symbol, **kw)
        self.cancel_threshold = cancel_threshold

    def score(self, snapshot: dict) -> float:
        cancel_rate = snapshot.get('cancel_rate', 0.0)
        obi         = abs(snapshot.get('obi', 0.0))
        raw = cancel_rate * 0.7 + obi * 0.3
        return self._normalize(raw)


# ─── Composed class using both ────────────────────────────────────
class AlertingSpoofer(AlertingMixin, SpoofingDetector):
    """MRO: AlertingSpoofer → AlertingMixin → SpoofingDetector → BaseDetector"""
    # No extra code needed. super() chain handles everything.
    pass
super() is not "call the parent" — it calls the next class in the MRO. This is why the cooperative mixin pattern works. AlertingMixin.detect calling super().detect() actually calls SpoofingDetector.detect, not BaseDetector.detect.

Polymorphism — one interface, many detectors

Polymorphism lets you treat SpoofingDetector, LayeringDetector, WashTradeDetector identically through a shared interface. The engine doesn't know which detector it's running — it just calls .detect().

duck typing __call__ operator overloading structural subtyping
engine.py — polymorphic detector pipeline 
class DetectionEngine:
    """
    Knows nothing about specific detector types.
    Works with ANY object that has a .detect(snapshot) method.
    This is Python duck typing — no interface declaration needed.
    """
    def __init__(self):
        self._detectors: list = []

    def register(self, detector) -> 'DetectionEngine':
        self._detectors.append(detector)
        return self  # fluent interface

    def run(self, snapshot: dict) -> list:
        alerts = []
        for det in self._detectors:
            alert = det.detect(snapshot)  # ← pure polymorphism
            if alert: alerts.append(alert)
        return alerts

    # Operator overloading: engine + detector = new engine
    def __add__(self, detector):
        self.register(detector)
        return self

    # Makes the engine callable: engine(snapshot)
    def __call__(self, snapshot: dict) -> list:
        return self.run(snapshot)

    def __len__(self): return len(self._detectors)
    def __repr__(self):
        names = [d.__class__.__name__ for d in self._detectors]
        return f"DetectionEngine({names})"


# ─── Usage ────────────────────────────────────────────────────────
engine = (
    DetectionEngine()
    + AlertingSpoofer("BTCUSDT")
    + LayeringDetector("BTCUSDT", depth=10)
    + WashTradeDetector("BTCUSDT", time_window=60)
)

# This custom lambda-style "detector" also works — duck typing
class VolumeSpike:
    def detect(self, snap):
        if snap.get('volume', 0) > 1_000_000:
            return f"VolumeSpike: {snap['volume']}"

engine + VolumeSpike()  # works! No inheritance required.

# Run — engine acts like a function
alerts = engine({
    'cancel_rate': 0.92,
    'obi': 0.68,
    'volume': 2_500_000
})
Duck typing vs nominal typing: Python doesn't require VolumeSpike to inherit from BaseDetector. It just needs a .detect() method. This is Python's structural subtyping. For production code with type checking, pair this with Protocol (next section).

Protocols & ABCs — define contracts

ABCs enforce a contract at instantiation time. Protocols enable structural typing — static type checkers verify compliance without inheritance. Critical for a multi-component system.

typing.Protocol ABC / abstractmethod runtime_checkable @overload
ABC — runtime enforcement
abc_pattern.py 
from abc import ABC, abstractmethod

class DataFeed(ABC):
    @abstractmethod
    async def subscribe(self,
            symbols: list[str]) -> None: ...

    @abstractmethod
    async def next_event(self) -> dict: ...

    @abstractmethod
    async def close(self) -> None: ...

# ❌ Raises TypeError at class definition
class BadFeed(DataFeed):
    async def subscribe(self, s): ...
    # forgot next_event and close
    # → TypeError: Can't instantiate
    #   abstract class BadFeed
Protocol — static typing only
protocol_pattern.py 
from typing import Protocol, runtime_checkable

@runtime_checkable
class Detector(Protocol):
    symbol: str

    def detect(self, snap: dict
              ) -> Optional['Alert']: ...

# ✅ No inheritance needed!
class MyDetector:
    symbol = "ETHUSDT"
    def detect(self, snap):
        return None

# isinstance check works because @runtime_checkable
isinstance(MyDetector(), Detector)  # → True

# mypy / pyright catches this statically:
def run(d: Detector) -> None: ...
run(MyDetector())   # ✅ passes type check
run("not a detector")  # ❌ type error

Descriptors — reusable attribute logic

Descriptors power @property, @classmethod, @staticmethod internally. Writing custom descriptors lets you attach validation, caching, or transformation logic to any attribute, once, reusable across every class.

__get__ / __set__ data vs non-data __set_name__ WeakKeyDictionary

1

3

descriptors.py — validated fields for trading params 
from weakref import WeakKeyDictionary

class BoundedFloat:
    """
    Descriptor: float that must stay in [lo, hi].
    Reuse on any class — no boilerplate.
    """
    def __init__(self, lo: float, hi: float):
        self.lo, self.hi = lo, hi
        self._data = WeakKeyDictionary()  # per-instance storage
        self._name = ''

    def __set_name__(self, owner, name):
        # Called at class creation: knows its own attribute name
        self._name = name

    def __get__(self, obj, objtype=None):
        if obj is None: return self        # class access → return descriptor
        return self._data.get(obj, 0.0)

    def __set__(self, obj, value: float):
        if not (self.lo <= value <= self.hi):
            raise ValueError(
                f"{self._name}={value} outside [{self.lo}, {self.hi}]")
        self._data[obj] = value


class PositiveInt:
    """Descriptor: integer > 0."""
    def __set_name__(self, owner, name):
        self._attr = '_' + name   # store as _window on instance dict

    def __get__(self, obj, _=None):
        if obj is None: return self
        return getattr(obj, self._attr, 1)

    def __set__(self, obj, val: int):
        if not isinstance(val, int) or val <= 0:
            raise TypeError(f"Must be int > 0, got {val!r}")
        setattr(obj, self._attr, val)


# ─── Apply descriptors declaratively ──────────────────────────────
class DetectorConfig:
    # These are class-level descriptors. No @property boilerplate per attr.
    alert_threshold = BoundedFloat(0.0, 1.0)
    cancel_threshold = BoundedFloat(0.0, 1.0)
    window = PositiveInt()
    depth  = PositiveInt()

    def __init__(self, **kw):
        for k, v in kw.items():
            setattr(self, k, v)  # all go through descriptors

cfg = DetectorConfig(alert_threshold=0.8, window=100)
cfg.alert_threshold = 1.5  # ❌ ValueError: outside [0.0, 1.0]
cfg.window = 0              # ❌ TypeError: Must be int > 0
Why WeakKeyDictionary? If you store instance data on the descriptor object itself (as a regular dict keyed by instance), the instance is never garbage collected — the descriptor holds a strong reference. WeakKeyDictionary uses weak references, so when an instance dies, its entry in the descriptor is cleaned up automatically.

Metaclasses — classes that build classes

A metaclass intercepts class creation. Use them to auto-register detectors, enforce naming conventions, inject logging, or build plugin systems without modifying individual classes.

type __init_subclass__ __new__ vs __init__ class registry pattern
Rule of thumb: Most of the time, use __init_subclass__ instead of a full metaclass. It was added in Python 3.6 specifically to cover the common metaclass use cases with much less complexity.
registry.py — auto-register detectors via __init_subclass__ 
class BaseDetector(ABC):
    # Class-level registry: name → class
    _registry: dict[str, type] = {}

    def __init_subclass__(cls, name: str = "", **kw):
        """
        Called automatically whenever someone subclasses BaseDetector.
        name= is passed as a keyword arg in the class statement.
        """
        super().__init_subclass__(**kw)
        key = name or cls.__name__
        if key in cls._registry:
            raise ValueError(f"Detector '{key}' already registered")
        cls._registry[key] = cls

    @classmethod
    def create(cls, name: str, **kw) -> 'BaseDetector':
        """Factory: instantiate any detector by name."""
        if name not in cls._registry:
            raise KeyError(f"Unknown detector: {name!r}. "
                           f"Available: {list(cls._registry)}")
        return cls._registry[name](**kw)


# ─── Subclasses auto-register ─────────────────────────────────────
class SpoofingDetector(BaseDetector, name="spoofing"):
    ...

class LayeringDetector(BaseDetector, name="layering"):
    ...

class WashTradeDetector(BaseDetector, name="wash_trade"):
    ...

# ─── Config-driven instantiation ──────────────────────────────────
config = [
    {"type": "spoofing",   "symbol": "BTCUSDT"},
    {"type": "layering",   "symbol": "ETHUSDT"},
    {"type": "wash_trade", "symbol": "SOLUSDT"},

meta.py — enforce naming conventions at class creation 
class DetectorMeta(type):
    """
    type is the metaclass of all classes.
    Subclass it to intercept class creation.
    """
    def __new__(mcs, name, bases, namespace):
        # Enforce: every non-abstract Detector must define 'KIND'
        if bases and 'KIND' not in namespace:
            raise TypeError(
                f"Detector subclass '{name}' must define KIND = 'spoofing' etc.")
        cls = super().__new__(mcs, name, bases, namespace)
        return cls

    def __init__(cls, name, bases, namespace):
        super().__init__(name, bases, namespace)
        # Auto-inject a class-level logger
        import logging
        cls.logger = logging.getLogger(f"vektor.{name}")


class BaseDetector(metaclass=DetectorMeta):
    KIND = "base"   # base defines it, subclasses must too

# ❌ TypeError at class definition time, not at runtime
class BadDetector(BaseDetector):
    pass  # forgot KIND → TypeError immediately

# ✅ Correct
class SpoofingDetector(BaseDetector):
    KIND = "spoofing"
    # self.logger is injected automatically
Metaclass conflicts: If two bases have different metaclasses, Python raises TypeError: metaclass conflict. This is why most libraries avoid metaclasses and use __init_subclass__ instead — it composes without conflicts.

Composition over Inheritance — the real pattern

A deep inheritance tree becomes brittle. The alternative: small, single-purpose objects composed together. This is how Django ORM, SQLAlchemy, FastAPI, and every solid Python framework is actually built.

dataclasses __post_init__ dependency injection __slots__ + frozen
vektor_core.py — full composition example 
from dataclasses import dataclass, field
from typing import Protocol

# ─── Value objects (immutable, frozen=True) ───────────────────────
@dataclass(frozen=True, slots=True)
class Snapshot:
    symbol:      str
    timestamp:   float
    best_bid:    float
    best_ask:    float
    obi:         float   # order book imbalance
    cancel_rate: float
    volume:      float

    def __post_init__(self):
        if self.best_bid >= self.best_ask:
            raise ValueError("bid >= ask: crossed book")
        if not (-1.0 <= self.obi <= 1.0):
            raise ValueError(f"obi {self.obi} out of [-1, 1]")


@dataclass(frozen=True, slots=True)
class Alert:
    symbol:    str
    score:     float
    kind:      str
    timestamp: float = field(default_factory=lambda: __import__('time').time())

    def __str__(self):
        return f"[{self.kind.upper()}] {self.symbol} score={self.score:.2f}"


# ─── Protocols as dependencies ────────────────────────────────────
class AlertSink(Protocol):
    async def send(self, alert: Alert) -> None: ...

class SnapshotSource(Protocol):
    async def __aiter__(self) -> AsyncIterator[Snapshot]: ...


# ─── Engine: composed of injected dependencies ────────────────────
class VektorEngine:
    """
    Doesn't inherit from anything business-logic-y.
    Composed of: source, detectors[], sinks[].
    Each can be swapped independently (test vs production).
    """
    def __init__(
        self,
        source:    SnapshotSource,
        detectors: list,          # list[Detector Protocol]
        sinks:     list[AlertSink]
    ):
        self._source    = source
        self._detectors = detectors
        self._sinks      = sinks

    async def run(self) -> None:
        async for snap in self._source:
            for det in self._detectors:
                alert = det.detect(snap)
                if alert:
                    for sink in self._sinks:
                        await sink.send(alert)


# ─── Wire it up ───────────────────────────────────────────────────
engine = VektorEngine(
    source=BinanceWebsocketFeed("BTCUSDT"),
    detectors=[
        AlertingSpoofer("BTCUSDT"),
        LayeringDetector("BTCUSDT"),
    ],
    sinks=[
        DiscordSink(webhook_url="..."),
        SQLiteSink(db_path="alerts.db"),
    ]
)

# In tests: swap BinanceFeed for MockFeed, DiscordSink for CaptureSink
# Zero code changes to VektorEngine itself.
This is the endgame pattern. VektorEngine has no knowledge of Binance, Discord, spoofing, or SQLite. It only knows three protocols: source, detectors, sinks. This makes it trivially testable and infinitely extensible — add a new exchange feed or new alert channel by writing a new class, not modifying existing ones. That's the Open/Closed Principle.
